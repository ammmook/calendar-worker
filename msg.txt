<script>
// ── State ──────────────────────────────────────────────────
let entries = {};
let otMult = 2;
let selectedKey = null;
const today = new Date();
let viewY = today.getFullYear();
let viewM = today.getMonth();

// ── Seed ───────────────────────────────────────────────────
function seed() {
  const y = today.getFullYear(), m = today.getMonth();
  const rows = [
    [1,'08:30','18:45'],[2,'09:00','17:00'],[3,'08:00','20:30'],
    [4,'09:15','18:00'],[5,'08:30','19:00'],[8,'09:00','21:00'],
    [9,'08:45','17:45'],[10,'09:00','22:00'],[11,'08:30','18:30'],
    [12,'09:00','17:00'],[15,'08:00','19:30'],[16,'09:00','20:00'],
    [17,'08:30','17:30'],[18,'09:00','18:00'],[19,'08:00','21:00'],
  ];
  rows.forEach(([d, inT, outT]) => {
    if (d > today.getDate()) return;
    const k = dateKey(y, m+1, d);
    entries[k] = { in: inT, out: outT };
  });
}

// ── Helpers ────────────────────────────────────────────────
function dateKey(y, m, d) {
  return `${y}-${String(m).padStart(2,'0')}-${String(d).padStart(2,'0')}`;
}

function todayKey() {
  return dateKey(today.getFullYear(), today.getMonth()+1, today.getDate());
}

function calc(inT, outT) {
  if (!inT || !outT) return { total:0, reg:0, ot:0 };
  const [ih,im]=inT.split(':').map(Number), [oh,om]=outT.split(':').map(Number);
  const mins = (oh*60+om)-(ih*60+im);
  if (mins <= 0) return { total:0, reg:0, ot:0 };
  const total = mins/60;
  const std = parseFloat(document.getElementById('std')?.value||8);
  const reg = Math.min(total, std);
  const ot = Math.max(0, total - std);
  return { total, reg, ot };
}

function earn(reg, ot) {
  const w = parseFloat(document.getElementById('wage')?.value||85);
  return (reg * w) + (ot * w * otMult);
}

function fmt1(n) { return n.toFixed(1); }
function fmtB(n) { return '฿' + Math.round(n).toLocaleString('en-US'); }
function fmtHrs(n) { return fmt1(n) + ' hrs'; }

// ── Filter population ──────────────────────────────────────
function populateFilters() {
  const MONTHS = ['January','February','March','April','May','June','July','August','September','October','November','December'];
  const ms = document.getElementById('filter-month');
  const ys = document.getElementById('filter-year');
  ms.innerHTML = MONTHS.map((mn,i) => `<option value="${i}"${i===viewM?' selected':''}>${mn}</option>`).join('');
  ys.innerHTML = '';
  for (let y = today.getFullYear()-2; y <= today.getFullYear()+1; y++) {
    const opt = document.createElement('option');
    opt.value = y; opt.textContent = y;
    if (y === viewY) opt.selected = true;
    ys.appendChild(opt);
  }
}

function onFilterChange() {
  viewM = parseInt(document.getElementById('filter-month').value);
  viewY = parseInt(document.getElementById('filter-year').value);
  renderCalendar();
  updateCards();
}

// ── Cards ──────────────────────────────────────────────────
function updateCards() {
  const keys = Object.keys(entries).filter(k => {
    const [y,m] = k.split('-').map(Number);
    return y === viewY && m === viewM+1;
  });

  let totalReg=0, totalOT=0, otDays=0, daysWorked=0;
  keys.forEach(k => {
    const e = entries[k];
    const h = calc(e.in, e.out);
    if (h.total > 0) { daysWorked++; totalReg += h.reg; totalOT += h.ot; if (h.ot > 0) otDays++; }
  });

  const regEarn = totalReg * parseFloat(document.getElementById('wage')?.value||85);
  const otEarn  = totalOT  * parseFloat(document.getElementById('wage')?.value||85) * otMult;
  const total   = regEarn + otEarn;

  document.getElementById('c-ot-hrs').textContent  = fmtHrs(totalOT);
  document.getElementById('c-ot-days').textContent = `${otDays} day${otDays!==1?'s':''} with overtime`;
  document.getElementById('c-ot-earn').textContent = fmtB(otEarn);
  document.getElementById('c-ot-rate').textContent = `at ${otMult}× multiplier`;
  document.getElementById('c-reg-earn').textContent= fmtB(regEarn);
  document.getElementById('c-reg-hrs').textContent = fmtHrs(totalReg) + ' regular';
  document.getElementById('c-total-earn').textContent = fmtB(total);
  document.getElementById('c-total-days').textContent  = `${daysWorked} day${daysWorked!==1?'s':''} worked`;

  const MONTHS = ['January','February','March','April','May','June','July','August','September','October','November','December'];
  document.getElementById('dash-title').textContent = 'Dashboard';
  document.getElementById('dash-subtitle').textContent = `${MONTHS[viewM]} ${viewY} — ${daysWorked} days worked, ${fmtHrs(totalOT)} overtime`;
}

// ── Calendar ───────────────────────────────────────────────
function renderCalendar() {
  const MONTHS = ['January','February','March','April','May','June','July','August','September','October','November','December'];
  document.getElementById('cal-label').textContent = `${MONTHS[viewM]} ${viewY}`;

  const grid = document.getElementById('cal-grid');
  grid.innerHTML = '';

  const firstDow = new Date(viewY, viewM, 1).getDay();
  const daysInM  = new Date(viewY, viewM+1, 0).getDate();
  const tk = todayKey();

  for (let i = 0; i < firstDow; i++) {
    const e = document.createElement('div');
    e.className = 'day-cell empty';
    grid.appendChild(e);
  }

  for (let d = 1; d <= daysInM; d++) {
    const k = dateKey(viewY, viewM+1, d);
    const entry = entries[k];
    const isToday = k === tk;
    const isSel   = k === selectedKey;
    const dow = new Date(viewY, viewM, d).getDay();
    const isWE = dow === 0 || dow === 6;

    let h = { total:0, reg:0, ot:0 };
    if (entry) h = calc(entry.in, entry.out);
    const hasOT = h.ot > 0;
    const earnings = earn(h.reg, h.ot);

    const cell = document.createElement('div');
    cell.className = [
      'day-cell',
      isToday ? 'today' : '',
      hasOT   ? 'has-ot' : '',
      isSel   ? 'selected' : '',
      isWE    ? 'weekend' : '',
    ].filter(Boolean).join(' ');

    cell.onclick = () => selectDay(k, d);

    let html = `<div class="day-num">${d}</div>`;
    if (hasOT) html += `<div class="ot-pip"></div>`;
    if (entry) {
      html += `<div class="day-info day-regular">${fmt1(h.total)}h</div>`;
      if (hasOT) html += `<div class="day-info day-ot-text">OT +${fmt1(h.ot)}h</div>`;
      html += `<div class="day-info day-earn">${fmtB(earnings)}</div>`;
    }

    cell.innerHTML = html;
    grid.appendChild(cell);
  }

  renderLogList();
  updateCards();
}

// ── Day Select ─────────────────────────────────────────────
function selectDay(key, d) {
  selectedKey = key;
  renderCalendar();

  const MONTHS=['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'];
  const DAYS=['Sun','Mon','Tue','Wed','Thu','Fri','Sat'];
  const dow = new Date(viewY, viewM, d).getDay();
  document.getElementById('panel-day-title').textContent = `${DAYS[dow]}, ${d} ${MONTHS[viewM]}`;

  const isToday = key === todayKey();
  document.getElementById('today-pill').style.display = isToday ? 'inline-block' : 'none';

  const entry = entries[key] || {};
  let h = { total:0, reg:0, ot:0 };
  if (entry.in && entry.out) h = calc(entry.in, entry.out);
  const e = earn(h.reg, h.ot);

  document.getElementById('day-detail-content').innerHTML = `
    <div class="time-row">
      <div class="time-group">
        <div class="time-label">Clock In</div>
        <input type="time" class="time-input" id="d-in" value="${entry.in||''}" oninput="liveCalc('${key}')">
      </div>
      <div class="time-group">
        <div class="time-label">Clock Out</div>
        <input type="time" class="time-input" id="d-out" value="${entry.out||''}" oninput="liveCalc('${key}')">
      </div>
    </div>
    <div class="calc-box">
      <div class="calc-row"><span class="calc-key">Total</span><span class="calc-val" id="d-total">${entry.in ? fmt1(h.total)+'h':'—'}</span></div>
      <hr class="calc-divider">
      <div class="calc-row"><span class="calc-key">Regular</span><span class="calc-val blue" id="d-reg">${entry.in ? fmt1(h.reg)+'h':'—'}</span></div>
      <div class="calc-row"><span class="calc-key">Overtime</span><span class="calc-val amber" id="d-ot">${entry.in ? fmt1(h.ot)+'h':'—'}</span></div>
      <hr class="calc-divider">
      <div class="calc-row"><span class="calc-key">Earnings</span><span class="calc-val green" id="d-earn">${entry.in ? fmtB(e):'—'}</span></div>
    </div>
    <button class="btn-save" onclick="saveDayEntry('${key}')">
      ${entry.in ? '✓ Update Entry' : '＋ Save Entry'}
    </button>
  `;
}

function liveCalc(key) {
  const inV  = document.getElementById('d-in')?.value;
  const outV = document.getElementById('d-out')?.value;
  if (!inV || !outV) return;
  const h = calc(inV, outV);
  const e = earn(h.reg, h.ot);
  const set = (id,v) => { const el = document.getElementById(id); if(el) el.textContent = v; };
  set('d-total', fmt1(h.total)+'h');
  set('d-reg',   fmt1(h.reg)+'h');
  set('d-ot',    fmt1(h.ot)+'h');
  set('d-earn',  fmtB(e));
}

function saveDayEntry(key) {
  const inV  = document.getElementById('d-in')?.value;
  const outV = document.getElementById('d-out')?.value;
  if (!inV || !outV) return toast('Please enter both times');
  entries[key] = { in: inV, out: outV };
  renderCalendar();
  selectDay(key, parseInt(key.split('-')[2]));
  toast('Entry saved ✓');
}

// ── Log List ───────────────────────────────────────────────
function renderLogList() {
  const DAYS=['Sun','Mon','Tue','Wed','Thu','Fri','Sat'];
  const keys = Object.keys(entries)
    .filter(k => { const [y,m]=k.split('-').map(Number); return y===viewY && m===viewM+1; })
    .sort((a,b) => b.localeCompare(a));

  document.getElementById('log-count').textContent = keys.length + ' entries';

  const list = document.getElementById('log-list');
  if (!keys.length) {
    list.innerHTML = '<div style="color:var(--ink-4); font-size:0.78rem; text-align:center; padding:12px 0;">No entries this month</div>';
    return;
  }

  list.innerHTML = keys.map(k => {
    const e = entries[k];
    const d = parseInt(k.split('-')[2]);
    const dow = new Date(k).getDay();
    const h = calc(e.in, e.out);
    const hasOT = h.ot > 0;
    return `
      <div class="log-item${hasOT?' log-ot':''}" onclick="selectDay('${k}',${d})">
        <div class="log-date-num">${String(d).padStart(2,'0')}</div>
        <div class="log-info">
          <div class="log-info-day">${DAYS[dow]}</div>
          <div class="log-info-time">${e.in} → ${e.out}</div>
        </div>
        <div class="log-right">
          <div class="log-hrs">${fmt1(h.total)}h</div>
          ${hasOT?`<span class="log-ot-tag">OT ${fmt1(h.ot)}h</span>`:''}
        </div>
      </div>`;
  }).join('');
}

function saveEntry() {
  const date = document.getElementById('m-date').value;
  const inV  = document.getElementById('m-in').value;
  const outV = document.getElementById('m-out').value;
  if (!date || !inV || !outV) return toast('Please fill all fields');
  entries[date] = { in: inV, out: outV };
  closeModal();

  viewY = y; viewM = m-1;
  populateFilters();
  renderCalendar();
  toast('Entry logged ✓');
}
